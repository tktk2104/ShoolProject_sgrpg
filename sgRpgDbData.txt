CREATE DATABASE sgrpg;
USE sgrpg;

CREATE TABLE user(
    id 	integer AUTO_INCREMENT,
    lv 	integer,
    exp 	integer,
    money 	integer,

    PRIMARY KEY(id) 
);

CREATE TABLE characterData(
    id 	integer AUTO_INCREMENT,
    name 	varchar(64),

    PRIMARY KEY(id)
);

CREATE TABLE userChara(
    id 		integer AUTO_INCREMENT,
    user_id 	integer,
    chara_id 	integer,

    PRIMARY KEY(id)
);
